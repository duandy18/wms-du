                                |         ), agg_picks AS (                                                                                                                                                                                      +

                                |           GROUP BY (((COALESCE(l.occurred_at, l.created_at) AT TIME ZONE 'utc'::text))::date), l.warehouse_id, COALESCE(( SELECT (ae.meta ->> 'platform'::text)                                                +
                                |                    FROM audit_events ae                                                                                                                                                                        +
                                |                   WHERE (((ae.category)::text = 'OUTBOUND'::text) AND ((ae.ref)::text = (l.ref)::text))                                                                                                        +
                                |                   ORDER BY ae.created_at DESC                                                                                                                                                                  +

                                |         )                                                                                                                                                                                                      +
